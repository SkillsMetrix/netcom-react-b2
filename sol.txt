import { StyleSheet, Text, View, TextInput, Button } from "react-native";
import React, { isValidElement, useState } from "react";
import * as SMS from "expo-sms";

export default function SmsApp() {
  const [number, setNumber] = useState("");
  const [message, setMessage] = useState("");

  const checkSMS = async () => {
    const isAvailable = await SMS.isAvailableAsync();
    if (isAvailable) {
      alert("SMS service is active");
    } else {
      alert("service not active");
    }
  };
  const sendSMS = async () => {
    const { result } = await SMS.sendSMSAsync(number, message);
    if (result === "sent") {
      console.log(result);
      alert("message sent");
    }
  };
  return (
    <View>
      <Text>SmsApp</Text>
      <TextInput
        value={message}
        placeholder="Enter Message"
        onChangeText={setMessage}
      />
      <TextInput
        value={number}
        placeholder="Enter Number"
        onChangeText={setNumber}
        keyboardType="phone-pad"
      />
      <Button title="Send SMS" onPress={sendSMS}></Button>

      <Button title="check Service" onPress={checkSMS}></Button>
    </View>
  );
}

const styles = StyleSheet.create({});
